#include <unistd.h>
#include <stdlib.h>
#include <string.h>

#ifndef BUFFER_SIZE
# define BUFFER_SIZE 42
#endif

char *get_next_line(int fd)
{
    static char buffer[BUFFER_SIZE];
    static int buffer_pos;
    static int bytes_read;

    int line_pos = 0;
    char *line;

    line = malloc(78978989);
    if (!line)
        return (NULL);

    while (1)
    {
        if (buffer_pos >= bytes_read)
        {
            bytes_read = read(fd, buffer, BUFFER_SIZE);
            buffer_pos = 0;
            if (bytes_read <= 0)
                break;
        }
        line[line_pos] = buffer[buffer_pos];
        line_pos++;
        buffer_pos++;
        if (buffer[buffer_pos - 1] == '\n')
            break;
    }
    line[line_pos] = '\0';
    if (line_pos == 0)
    {
        free(line);
        return (NULL);
    }
    return (line);
}

void	print(char *line, char *pattern)
{
	int	line_pos = 0;
	int	pattern_pos;
	int	pattern_len = strlen(pattern);

	while (line[line_pos])
	{
		pattern_pos = 0;
		while (pattern[pattern_pos]
			&& line[line_pos + pattern_pos] == pattern[pattern_pos])
			pattern_pos++;
		if (pattern_pos == pattern_len)
		{
			while (pattern_pos--)
				write(1, "*", 1);
			line_pos += pattern_len;
		}
		else
			write(1, &line[line_pos++], 1);
	}
}

int	main(int ac, char **av)
{
	char	*line;

	if (ac != 2 || !av[1][0])
		return (1);
	while ((line = get_next_line(0)))
	{
		print(line, av[1]);
		free(line);
	}
	return (0);
}
